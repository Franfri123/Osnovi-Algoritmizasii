using System;

namespace TicTacToeGame
{
    class Program
    {
        static char[,] board =
        {
            { '1', '2', '3' },
            { '4', '5', '6' },
            { '7', '8', '9' }
        };

        static int turns = 0;

        static void Main(string[] args)
        {
            char currentPlayer = 'X';
            bool gameEnded = false;

            while (!gameEnded)
            {
                DrawBoard();

                if (currentPlayer == 'X')
                {
                    PlayerMove(currentPlayer);
                }
                else
                {
                    Console.WriteLine("\nХод бота (O)...");
                    System.Threading.Thread.Sleep(1000); // Пауза для реалистичности
                    AiMove(currentPlayer);
                }

                if (CheckWin(currentPlayer))
                {
                    DrawBoard();
                    Console.WriteLine(currentPlayer == 'X' ? "\nПоздравляем! Вы победили!" : "\nБот победил!");
                    gameEnded = true;
                }
                else if (turns == 9)
                {
                    DrawBoard();
                    Console.WriteLine("\nНичья!");
                    gameEnded = true;
                }
                else
                {
                    currentPlayer = (currentPlayer == 'X') ? 'O' : 'X';
                }
            }
            Console.ReadKey();
        }

        static void DrawBoard()
        {
            Console.Clear();
            Console.WriteLine("Крестики-Нолики против БОТА");
            Console.WriteLine("-------------------------------------");
            Console.WriteLine($" {board[0, 0]} | {board[0, 1]} | {board[0, 2]} ");
            Console.WriteLine("---|---|---");
            Console.WriteLine($" {board[1, 0]} | {board[1, 1]} | {board[1, 2]} ");
            Console.WriteLine("---|---|---");
            Console.WriteLine($" {board[2, 0]} | {board[2, 1]} | {board[2, 2]} ");
            Console.WriteLine("-------------------------------------");
        }

        static void PlayerMove(char playerSymbol)
        {
            bool validMove = false;
            while (!validMove)
            {
                Console.Write($"\nВаш ход ({playerSymbol}). Клетка (1-9): ");
                string input = Console.ReadLine();

                if (int.TryParse(input, out int cell) && cell >= 1 && cell <= 9)
                {
                    int row = (cell - 1) / 3;
                    int col = (cell - 1) % 3;

                    if (board[row, col] != 'X' && board[row, col] != 'O')
                    {
                        board[row, col] = playerSymbol;
                        turns++;
                        validMove = true;
                    }
                    else Console.WriteLine("Занято!");
                }
                else Console.WriteLine("Введите число 1-9!");
            }
        }

        // Логика ИИ
        static void AiMove(char botSymbol)
        {
            char opponentSymbol = 'X';

            // 1. Пытаемся победить, если есть возможность
            if (TryFindBestMove(botSymbol, out int r1, out int c1)) { ApplyMove(r1, c1, botSymbol); return; }

            // 2. Блокируем игрока, если он может победить следующим ходом
            if (TryFindBestMove(opponentSymbol, out int r2, out int c2)) { ApplyMove(r2, c2, botSymbol); return; }

            // 3. Если центр свободен — берем его
            if (board[1, 1] != 'X' && board[1, 1] != 'O') { ApplyMove(1, 1, botSymbol); return; }

            // 4. Иначе — случайный ход
            Random rand = new Random();
            while (true)
            {
                int r = rand.Next(0, 3);
                int c = rand.Next(0, 3);
                if (board[r, c] != 'X' && board[r, c] != 'O')
                {
                    ApplyMove(r, c, botSymbol);
                    break;
                }
            }
        }

        static bool TryFindBestMove(char symbol, out int row, out int col)
        {
            for (int i = 0; i < 9; i++)
            {
                int r = i / 3;
                int c = i % 3;
                char original = board[r, c];

                if (original != 'X' && original != 'O')
                {
                    board[r, c] = symbol; // Временно ставим символ
                    if (CheckWin(symbol))
                    {
                        board[r, c] = original; // Возвращаем назад
                        row = r; col = c;
                        return true;
                    }
                    board[r, c] = original;
                }
            }
            row = -1; col = -1;
            return false;
        }

        static void ApplyMove(int r, int c, char symbol)
        {
            board[r, c] = symbol;
            turns++;
        }

        static bool CheckWin(char s)
        {
            for (int i = 0; i < 3; i++)
            {
                if ((board[i, 0] == s && board[i, 1] == s && board[i, 2] == s) ||
                    (board[0, i] == s && board[1, i] == s && board[2, i] == s)) return true;
            }
            return (board[0, 0] == s && board[1, 1] == s && board[2, 2] == s) ||
                   (board[0, 2] == s && board[1, 1] == s && board[2, 0] == s);
        }
    }
}
