Вопрос 1

Какие преобразования типов НЕ выполняются автоматически:

  

1. Из int в short

2. Из short в int

3. Из bool в string

4. Из byte в float

Ответ:  1,2

```cs
using System;

  

class Program

{

    static void Main()

    {

        Console.WriteLine("Введите номер операции: 1.Сложение 2.Вычитание 3.Умножение");

  

        // преобразовываем в целое число (int)

        // если пользователь введет не число.

        if (int.TryParse(Console.ReadLine(), out int operationNumber))

        {

            // используем switch для проверки номера операции

            switch (operationNumber)

            {

                case 1:

                    Console.WriteLine("Выбрана операция: Сложение");

                    break; // break обязателен для выхода из switch после выполнения нужного блока

  

                case 2:

                    Console.WriteLine("Выбрана операция: Вычитание");

                    break;

  

                case 3:

                    Console.WriteLine("Выбрана операция: Умножение");

                    break;

  

                default:

                    // блок default выполняется, если operationNumber не совпал ни с одним из case

                    Console.WriteLine("Ошибка: Операция неопределена");

                    break;

            }

        }

        else

        {

            Console.WriteLine("Ошибка ввода: Введено некорректное число.");

        }

    }

}
```

**Упражнение 7**

```cs

using System;

  

class Program

{

    static void Main()

    {

        Console.WriteLine("Введите номер операции: 1.Сложение 2.Вычитание 3.Умножение");

  

        // считываем номер операции

        if (!int.TryParse(Console.ReadLine(), out int operationNumber))

        {

            Console.WriteLine("Ошибка: Введено некорректное значение для номера операции.");

            return;

        }

  

        Console.Write("Введите первое число: ");

        // считываем первое число

        if (!double.TryParse(Console.ReadLine(), out double numA))

        {

            Console.WriteLine("Ошибка: Введено некорректное значение для первого числа.");

            return;

        }

  

        Console.Write("Введите второе число: ");

        // считываем второе число

        if (!double.TryParse(Console.ReadLine(), out double numB))

        {

            Console.WriteLine("Ошибка: Введено некорректное значение для второго числа.");

            return;

        }

  

        double result;

  

        // выполняем операцию в зависимости от номера

        switch (operationNumber)

        {

            case 1:

                result = numA + numB;

                Console.WriteLine($"\nРезультат сложения: {numA} + {numB} = {result}");

                break;

  

            case 2:

                result = numA - numB;

                Console.WriteLine($"\nРезультат вычитания: {numA} - {numB} = {result}");

                break;

  

            case 3:

                result = numA * numB;

                Console.WriteLine($"\nРезультат умножения: {numA} * {numB} = {result}");

                break;

  

            default:

                Console.WriteLine("\nОшибка: Операция неопределена.");

                break;

        }

    }

}
```

**Практическая**

```cs

using System;
using System.Threading;

class Program
{
    static void Main()
    {
        // отслеживание наличия предмета, влияющего на концовки
        bool hasOrb = false; // В новой теме это "Энергоядро"

        Console.WriteLine("TERMINAL: ОБЪЕКТ «ГЕЛИОС-7»");
        Console.WriteLine("Вы стоите перед шлюзом заброшенной станции. Ваша цель — главный блок данных.");

    // шаг 1: вход в пещеру
    Step1:
        Console.WriteLine("\nВы у внешнего люка. В вакууме царит тишина.");
        Console.WriteLine("1. Начать декомпрессию и войти.");
        Console.WriteLine("2. Прервать миссию и вернуться на шаттл.");

        string choice = Console.ReadLine();

        if (choice == "1") goto Step2;
        else if (choice == "2") goto Ending_Coward; // концовка 1
        else { Console.WriteLine("Ошибка системы."); goto Step1; }

    // шаг 2: главный зал
    Step2:
        Console.WriteLine("\nВнутри — отсек реактора. В центре тускло мигает резервный генератор.");
        Console.WriteLine("1. Извлечь энергоядро.");
        Console.WriteLine("2. Пройти в технический тоннель (Сектор А).");
        Console.WriteLine("3. Пройти в жилой блок (Сектор Б).");

        choice = Console.ReadLine();

        if (choice == "1") goto Step3;
        else if (choice == "2") goto Step4;
        else if (choice == "3") goto Step5;
        else { Console.WriteLine("Ошибка системы."); goto Step2; }

    // шаг 3: кристалл
    Step3:
        Console.WriteLine("\nВы извлекаете ядро. Системы освещения гаснут, в руках — заряженный элемент.");
        hasOrb = true;
        Thread.Sleep(1500);
        goto Step2; // возвращаемся в главный зал

    // шаг 4: левый туннель (безопасный путь)
    Step4:
        Console.WriteLine("\nТехнический тоннель ведет вас к шлюзу внутренней палубы.");
        goto Step6;

    // шаг 5: правый туннель (Опасность)
    Step5:
        Console.WriteLine("\nВ жилом блоке происходит разгерметизация! Вы едва успеваете закрыть дверь за собой.");
        goto Step6;

    // шаг 6: подземное озеро
    Step6:
        Console.WriteLine("\nВы в отсеке жизнеобеспечения. В резервуаре с водой что-то плавает.");
        Console.WriteLine("1. Проверить содержимое резервуара.");
        Console.WriteLine("2. Двигаться дальше к мостику.");

        choice = Console.ReadLine();

        if (choice == "1") goto Step7;
        else if (choice == "2") goto Step8;
        else { Console.WriteLine("Ошибка системы."); goto Step6; }

    // шаг 7: поиск в воде (Найти второй предмет)
    Step7:
        Console.WriteLine("\nВы находите на дне забытый кем-то сервисный ключ доступа.");
        goto Step8;

    // шаг 8: другая сторона озера
    Step8:
        Console.WriteLine("\nВы добрались до развилки перед командным центром. Впереди две консоли: синяя и красная.");
        Console.WriteLine("1. Активировать синюю консоль.");
        Console.WriteLine("2. Активировать красную консоль.");

        choice = Console.ReadLine();

        if (choice == "1") goto Step9;
        else if (choice == "2") goto Step10;
        else { Console.WriteLine("Ошибка системы."); goto Step8; }

    // шаг 9: деревянная дверь (Ловушка)
    Step9:
        Console.WriteLine("\nСиняя консоль перегружает систему жизнеобеспечения. Воздух мгновенно откачивается.");
        goto Ending_Void; // концовка 2

    // шаг 10: металлическая дверь (Комната Стража)
    Step10:
        Console.WriteLine("\nКрасная консоль открывает дверь, за которой стоит боевой дрон в режиме ожидания.");
        Console.WriteLine("1. Попытаться бесшумно обойти датчики движения.");
        Console.WriteLine("2. Попробовать взломать систему дрона напрямую.");

        choice = Console.ReadLine();

        if (choice == "1") goto Step11;
        else if (choice == "2") goto Step12;
        else { Console.WriteLine("Ошибка системы."); goto Step10; }

    // шаг 11: прокрасться мимо
    Step11:
        Console.WriteLine("\nВам удается пройти в слепой зоне камер. Вы видите серверную.");
        goto Step13;

    // шаг 12: разбудить стража
    Step12:
        Console.WriteLine("\nСистема защиты просыпается! Дрон активирует турели. Сопротивление бесполезно.");
        goto Ending_Captured; // концовка 3

    // шаг 13: финальный зал
    Step13:
        Console.WriteLine("\nВы в серверной. Вот он — блок данных, но терминал заблокирован мощным файрволом.");
        goto Step14;

    // шаг 14: барьер
    Step14:
        Console.WriteLine("\nФайрволу требуется дополнительное питание для обхода защиты.");
        if (hasOrb)
        {
            Console.WriteLine("Вы подключаете энергоядро! Энергия сжигает протоколы защиты.");
            goto Step15;
        }
        else
        {
            Console.WriteLine("У вас нет источника питания. Станция блокирует все выходы. Вы в ловушке.");
            goto Ending_Trapped; // концовка 4
        }

    // шаг 15: Артефакт
    Step15:
        Console.WriteLine("\n[ШАГ 15] Вы скачиваете данные. Станция запускает протокол самоликвидации, но вы успеваете к шаттлу.");
        goto Ending_Victory; // концовка 5


    // концовки

    Ending_Coward:
        Console.WriteLine("\nОТКАЗ");
        Console.WriteLine("Вы покинули орбиту. Корпорация не прощает провалов, миссия забыта.");
        goto End;

    Ending_Void:
        Console.WriteLine("\nКРИТИЧЕСКАЯ ОШИБКА");
        Console.WriteLine("Вы погибли в вакууме космоса.");
        goto End;

    Ending_Captured:
        Console.WriteLine("\nОБЪЕКТ НЕЙТРАЛИЗОВАН");
        Console.WriteLine("Система безопасности поместила вас в криокамеру до прибытия властей.");
        goto End;

    Ending_Trapped:
        Console.WriteLine("\nИЗОЛЯЦИЯ");
        Console.WriteLine("Ресурсы скафандра на исходе. Вы останетесь в этой серверной навечно.");
        goto End;

    Ending_Victory:
        Console.WriteLine("\nМИССИЯ ВЫПОЛНЕНА");
        Console.WriteLine("Данные у вас. Вы успешно покинули «Гелиос-7». Награда ждет вас.");
        goto End;

    End:
        Console.WriteLine("\nСВЯЗЬ ПРЕРВАНА.");
    }
}
```
